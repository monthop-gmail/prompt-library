            except (UnicodeDecodeError, pd.errors.ParserError):

        raise ValueError("‡πÑ‡∏°‡πà‡∏™‡∏≤‡∏°‡∏≤‡∏£‡∏ñ decode ‡πÑ‡∏ü‡∏•‡πå‡πÑ‡∏î‡πâ")

                elif fmt.upper() in ("XLS", "XLSX"):
                    return self.download_excel(url)
        raise FileNotFoundError(f"‡πÑ‡∏°‡πà‡∏û‡∏ö resource format {fmt}")

    def query_datastore(self, resource_id: str,
                        filters: dict = None,
                        limit: int = 100) -> List[Dict]:
        """Query ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å CKAN Datastore"""

                            save_dir: str = "./data") -> List[str]:
