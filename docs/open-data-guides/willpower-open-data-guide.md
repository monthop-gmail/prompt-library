    def get_branch(self, branch_id: int) -> Optional[Branch]:
        """
        à¸”à¸¶à¸‡à¸‚à¹‰à¸­à¸¡à¸¹à¸¥à¸ªà¸²à¸‚à¸² 1 à¸ªà¸²à¸‚à¸²
        à¹€à¸—à¸µà¸¢à¸š: CKAN package_show

        # Parse à¸‚à¹‰à¸­à¸¡à¸¹à¸¥à¸ˆà¸²à¸ HTML
